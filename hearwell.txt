<script>
  const statusText = document.getElementById("status");
  const resultCard = document.getElementById("result");
  const resultText = document.getElementById("resultText");
  const precautionsBox = document.getElementById("precautions");
  const dateField = document.getElementById("date");

  const freqs = [500, 1000, 2000, 4000];
  let responses = {};
  let currentFreq = null;
  let testRunning = false;
  
  function say(msg){
  const s = new SpeechSynthesisUtterance(msg);
  s.lang="en-IN"; 
  s.rate=1.0; // medium pace voice
  window.speechSynthesis.speak(s);
}


  function playTone(freq, duration=3){
    const ctx = new (window.AudioContext || window.webkitAudioContext)();
    const osc = ctx.createOscillator();
    const gain = ctx.createGain();
    osc.frequency.value = freq;
    osc.type = "sine";
    osc.connect(gain);
    gain.connect(ctx.destination);
    gain.gain.setValueAtTime(0.3, ctx.currentTime);
    osc.start();
    osc.stop(ctx.currentTime + duration);
  }

  async function runTest(){
    if(testRunning) return;
    testRunning = true;
    responses = {};
    resultCard.style.display="none";
    statusText.textContent = "Starting hearing test...";

    for (let i=0; i<freqs.length; i++){
      const f = freqs[i];
      currentFreq = f;
      responses[f] = true; // assume heard, will change if user clicks can't
      await new Promise(r=>setTimeout(r,1000)); 
      statusText.textContent = `🔊 Playing ${f} Hz tone...`;
      playTone(f,3);
      await new Promise(r=>setTimeout(r,3500));
    }

    showResult();
    testRunning = false;
  }

  function cannotHear(){
    if(currentFreq){
      responses[currentFreq] = false;
      statusText.textContent = `❗ You reported that you can't hear ${currentFreq} Hz tone.`;
    }
  }

  function showResult(){
    statusText.textContent = "✅ Test complete!";
    resultCard.style.display = "block";
    dateField.textContent = new Date().toLocaleString();

    const heardAll = Object.values(responses).every(v => v === true);

    if(heardAll){
      resultText.innerHTML = "🎉 <b>Excellent hearing detected!</b>";
      resultText.className="good";
      precautionsBox.innerHTML = `
        <ul>
          <li>Avoid exposure to loud sounds for long durations.</li>
          <li>Use ear protection in noisy environments.</li>
          <li>Maintain clean ears and healthy habits.</li>
        </ul>`;
      say("Excellent hearing detected. Great job!");
    } else {
      resultText.innerHTML = "⚠️ <b>Possible hearing difficulty detected.</b>";
      resultText.className="bad";
      precautionsBox.innerHTML = `
        <ul>
          <li>Reduce earphone or loudspeaker volume.</li>
          <li>Avoid inserting objects into your ears.</li>
          <li>Consult an ENT doctor for a detailed checkup.</li>
        </ul>`;
      say("Possible hearing difficulty detected. Please consult an ear specialist soon.");
    }
  }

  document.getElementById("startTest").addEventListener("click", runTest);
  document.getElementById("cantHear").addEventListener("click", cannotHear);
</script>
